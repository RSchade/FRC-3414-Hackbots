002package edu.wpi.first.wpilibj.templates;
003
004import edu.wpi.first.wpilibj.SimpleRobot;
005import edu.wpi.first.wpilibj.Timer;
006import farmington.frameworks.CompressorControl;
007import farmington.frameworks.LoopHandler;
008import farmington.ultimateascent.Autonomous;
009import farmington.ultimateascent.IRobot;
010import farmington.ultimateascent.OperatorControl;
011
012public class Main extends SimpleRobot implements IRobot {
013    
014    Autonomous myAutonomous;
015    OperatorControl myOpControl;
016    CompressorControl myCompressorControl;
017    
018    public void autonomous() {
019        while (isAutonomous() && isEnabled()) {
020            myAutonomous.aim();
021            Timer.delay(TIME_DELAY);
022        }
023    }
024
025    public void operatorControl() {
026        
027        myOpControl = new OperatorControl();
028        int loopCount = 0;
029        
030        while (isOperatorControl() && isEnabled()) {
031            LoopHandler.updateLoopCount(loopCount);
032            
033            //Runs every 20 milliseconds
034            if (loopCount%2 == 0) {
035                myOpControl.twentyMSLoop();
036            }
037            
038            //Runs every 100 milliseconds
039            if (loopCount%10 == 0) {
040                myOpControl.hundredMSLoop();
041            }
042            
043            //Runs every 1000 milliseconds (every second)
044            if (loopCount%100 == 0) {
045                myOpControl.thousandMSLoop();
046            }
047            
048            loopCount++;
049            Timer.delay(TIME_DELAY);
050        }
051    }
052    
053    public void disabled() {
054        myOpControl.free();
055    }
056    
057    public void test() {
058        myCompressorControl = new CompressorControl(DIO_FOURTEEN, RELAY_TWO);
059        myCompressorControl.runCompressor();
060    }
061}
