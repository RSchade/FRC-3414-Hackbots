020public class Alexsstuff extends SimpleRobot implements IRobot
021{
022
023    private Joystick right_stick = new Joystick(1);
024    private Joystick left_stick = new Joystick(2);
025    private Talon firewheel = new Talon(PWM_SLOT_ONE);
026    private Talon firewheel_2 = new Talon(PWM_SLOT_TWO);
027    private Talon angler = new Talon(PWM_SLOT_THREE);
028    private Talon loaderwheel_1 = new Talon(PWM_SLOT_FOUR);
029    private Talon loaderwheel_2 = new Talon(PWM_SLOT_FIVE);
030    private Encoder anglereader[];
031    
032    private final int CENTER_BUTTON = 3;
033    private final int TRIGGER = 1;
034    private final int LEFT_BUTTON = 4;
035    private final int RIGHT_BUTTON = 5;
036    private final int OFF = 0;
037    private int SHOOTER_ANGLE_VALUE;
038    private final int BOTTOM_OUT = 40;  //These constants need 
039    private final int TOP_OUT = -40;    //to be fine tuned 
040    private final int FULL_FORWARD = 1;
041    private final int FULL_REVERSE = -1;
042    
043    private final double HALF_FORWARD = 0.50;
044    private final double HALF_REVERSE = -0.50;
045
046    public void autonomous()
047    {
048        
049    }
050
051    public void operatorControl()
052    {
053
054        while (isOperatorControl() && isEnabled())
055        {
056            controls();
057            angle();   
058            Timer.delay(0.010);
059            /*
060             * These incorprate the controls method
061             * and the angle limiting method and
062             * update every 10 miliseconds
063             */
064        }
065
066    }
067    
068    private void controls()
069    {
070        if (right_stick.getRawButton(TRIGGER) == true)
071        {
072            System.out.println("Fire!");
073            firewheel.set(FULL_FORWARD);
074            firewheel_2.set(FULL_FORWARD);
075        } else {
076            firewheel.set(OFF);
077            firewheel_2.set(OFF);
078        }
079
080        if (left_stick.getRawButton(TRIGGER) == true)
081        {
082            loaderwheel_1.set(HALF_FORWARD);
083            loaderwheel_2.set(HALF_FORWARD);
084        } 
085        else {
086            loaderwheel_1.set(OFF);
087            loaderwheel_2.set(OFF);
088        }
089
090        if (right_stick.getRawButton(CENTER_BUTTON) == true)
091        {
092            angler.set(FULL_FORWARD);
093            System.out.println("Raising!");
094        }
095
096        else  {
097            angler.set(OFF);
098        }
099
100        if (left_stick.getRawButton(CENTER_BUTTON) == true)
101        {
102            angler.set(FULL_REVERSE);
103            System.out.println("Lowering!");
104        }
105
106        if (left_stick.getRawButton(CENTER_BUTTON) == false
107                && right_stick.getRawButton(CENTER_BUTTON) == false)
108        {
109            angler.set(OFF);
110        }/*
111         * this thing just check what buttons are
112         * clicked, and activate what they need
113         */
114
115    }
116    
117    private void angle()
118    {
119        
120        anglereader = new Encoder[SHOOTER_ANGLE_VALUE];
121        
122        System.out.println(SHOOTER_ANGLE_VALUE);
123        
124        if(SHOOTER_ANGLE_VALUE -10 >= TOP_OUT)
125        {
126            angler.set(OFF);
127        }
128        if(SHOOTER_ANGLE_VALUE +10 <= BOTTOM_OUT)
129        {
130            angler.set(OFF);
131        }/*
132         * Checks to make sure the encoder
133         * value is within safe limits
134         */        
135    }
