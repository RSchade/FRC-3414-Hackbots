006package deprecated;
007
008import com.sun.squawk.util.MathUtils;
009import edu.wpi.first.wpilibj.AnalogChannel;
010import edu.wpi.first.wpilibj.PIDSource;
011import edu.wpi.first.wpilibj.SensorBase;
012
013/**
014 *
015 * @author charris
016 */
017public class Angle extends SensorBase implements PIDSource {
018
019    private final double BIAS_VOLTAGE = 2.25;
020    
021    private AnalogChannel sinChannel;
022    private AnalogChannel cosChannel;
023
024    public Angle( int slotParam, int sinChannelParam, int cosChannelParam)
025    {
026        sinChannel = new AnalogChannel( slotParam, sinChannelParam );
027        cosChannel = new AnalogChannel( slotParam, cosChannelParam );
028    }
029
030    protected double getCos()
031    {
032        return cosChannel.getVoltage() - BIAS_VOLTAGE;
033    }
034
035    protected double getSin()
036    {
037        return sinChannel.getVoltage() - BIAS_VOLTAGE;
038    }
039
040    public double getAngle()
041    {
042        // calculate the angle
043        double theta = MathUtils.atan2( getSin(), getCos() );
044        // convert to degrees
045        return Math.toDegrees(theta) + 180.0;
046    }
047
048    /**
049     * Get the angle of the encoder for use with PIDControllers
050     * @return the current angle according to the encoder
051     */
052    public double pidGet() 
053    {
054        return getAngle();
055    }
056}
